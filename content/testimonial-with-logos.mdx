<script>
  document.addEventListener("DOMContentLoaded", function () {
    const testimonialBox = document.getElementById(
      "convert-cat-tst-testimonial"
    );
    const logos = document.querySelectorAll(
      ".convert-cat-tst-logos-container img"
    );

    logos.forEach((logo) => {
      logo.addEventListener("click", () => {
        // Update testimonial text
        testimonialBox.textContent = '"' + logo.dataset.testimonial + '"';

        // Update active state
        logos.forEach((l) => (l.dataset.active = "false"));
        logo.dataset.active = "true";
      });
    });
  });
</script>
